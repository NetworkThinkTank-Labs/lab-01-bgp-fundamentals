! ---- eBGP Peering Template ----
router bgp <LOCAL_ASN>
 neighbor <NEIGHBOR_IP> remote-as <REMOTE_ASN>
 neighbor <NEIGHBOR_IP> description eBGP-to-<PEER_NAME>
 neighbor <NEIGHBOR_IP> password <MD5_KEY>
 neighbor <NEIGHBOR_IP> timers 10 30
 neighbor <NEIGHBOR_IP> fall-over
 !
 address-family ipv4 unicast
  neighbor <NEIGHBOR_IP> activate
  neighbor <NEIGHBOR_IP> soft-reconfiguration inbound
  neighbor <NEIGHBOR_IP> prefix-list <PREFIX_LIST_NAME> in
  neighbor <NEIGHBOR_IP> route-map <ROUTE_MAP_NAME> in
  neighbor <NEIGHBOR_IP> maximum-prefix 1000 80
 exit-address-family

! ---- iBGP Peering Template ----
router bgp <LOCAL_ASN>
 neighbor <NEIGHBOR_LOOPBACK> remote-as <LOCAL_ASN>
 neighbor <NEIGHBOR_LOOPBACK> description iBGP-to-<PEER_NAME>
 neighbor <NEIGHBOR_LOOPBACK> update-source Loopback0
 neighbor <NEIGHBOR_LOOPBACK> password <MD5_KEY>
 neighbor <NEIGHBOR_LOOPBACK> next-hop-self
 !
 address-family ipv4 unicast
  neighbor <NEIGHBOR_LOOPBACK> activate
  neighbor <NEIGHBOR_LOOPBACK> soft-reconfiguration inbound
 exit-address-family
